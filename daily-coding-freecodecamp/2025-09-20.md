def number_of_files(file_size, unit, drive_capacity_gb):
    drive_capacity_bytes = drive_capacity_gb * 1000 * 1000 * 1000

    if unit == "B":
        file_size_bytes = file_size
    elif unit == "KB":
        file_size_bytes = file_size * 1000
    elif unit == "MB":
        file_size_bytes = file_size * 1000 * 1000
    else:
        raise ValueError("Unit tidak valid. Gunakan 'B', 'KB', atau 'MB'.")

    return int(drive_capacity_bytes // file_size_bytes)
