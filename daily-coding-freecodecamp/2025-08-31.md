import random

def generate_hex(color):
    if color not in ["red", "green", "blue"]:
        return "Invalid color"

    # Membuat nilai acak untuk setiap channel warna (0 - 255)
    r = random.randint(0, 255)
    g = random.randint(0, 255)
    b = random.randint(0, 255)

    # Menentukan channel warna utama menjadi dominan (selalu 255)
    if color == "red":
        r = 255
    elif color == "green":
        g = 255
    elif color == "blue":
        b = 255

    # Mengembalikan kode warna hex (6 karakter, huruf kapital)
    return f"{r:02X}{g:02X}{b:02X}"
