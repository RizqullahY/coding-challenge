def get_laptop_cost(prices, budget):
    print('============')
    # 1. REMOVE DUPLICATE
    unique_prices = set(prices)
    
    # 2. TAKE PRICE
    low_price = [p for p in unique_prices if p <= budget]
    print(f'HIGH PRICE : {low_price}')
    
    # 3. TRUE?
    if not low_price:
        return 0
    
    # 4. SORT
    low_price.sort(reverse=True)
    print(f'HIGH PRICE REVERSED : {low_price}')

    return low_price[1] if len(low_price) > 1 else low_price[0]
